_ILLEGAL_XML_CHARS_RE = re.compile(r"[\x00-\x08\x0b\x0c\x0e-\x1f\x7f\ufffe\uffff]")

def _sanitize_text(text) -> str:
    if text is None:
        return ""
    if not isinstance(text, str):
        text = str(text)
    return _ILLEGAL_XML_CHARS_RE.sub("", text)

def get_space_title(space_key: str) -> str:
    url = f"{BASE_URL}/rest/api/space/{space_key}"
    try:
        r = requests.get(url, headers=HEADERS, verify=CERT_PATH, timeout=10)
        return r.json().get("name", space_key)
    except Exception:
        return space_key

def get_all_pages(space_key: str) -> list[dict]:
    """Fetch all space pages using the paginated export_view API endpoint."""
    pages, start, batch = [], 0, 50
    while True:
        url = f"{BASE_URL}/rest/api/content"
        params = {
            "spaceKey": space_key,
            "type": "page",
            "status": "current",
            "expand": "body.export_view,ancestors",
            "start": start,
            "limit": batch,
        }
        r = requests.get(url, headers=HEADERS, params=params, verify=CERT_PATH, timeout=20)
        data = r.json()
        results = data.get("results", [])
        pages.extend(results)
        
        if len(results) < batch:
            break
        start += batch
    return pages

def get_root_pages(space_key: str) -> list[dict]:
    """Scans and retrieves only top-level (root) pages for the UI dropdown selection."""
    url = f"{BASE_URL}/rest/api/content"
    params = {
        "spaceKey": space_key,
        "type": "page",
        "status": "current",
        "expand": "ancestors",
        "limit": 999
    }
    try:
        r = requests.get(url, headers=HEADERS, params=params, verify=CERT_PATH, timeout=20)
        all_content = r.json().get("results", [])
        # A root page has zero ancestors in Confluence
        roots = [{"id": p["id"], "title": p["title"]} for p in all_content if len(p.get("ancestors", [])) == 0]
        return sorted(roots, key=lambda x: x["title"])
    except Exception as e:
        print(f"[ERROR scanning root pages]: {e}")
        return []

def filter_page_tree(all_pages: list[dict], selected_parent_ids: list[str]) -> list[dict]:
    """Filters out any pages that do not belong to the selected parent hierarchies."""
    if not selected_parent_ids or "ALL" in selected_parent_ids:
        return all_pages
        
    set_parent_ids = set(selected_parent_ids)
    filtered = []
    
    for page in all_pages:
        page_id = page.get("id")
        ancestor_ids = {anc.get("id") for anc in page.get("ancestors", [])}
        
        # Keep page if it IS a selected parent or a CHILD of a selected parent
        if page_id in set_parent_ids or not ancestor_ids.isdisjoint(set_parent_ids):
            filtered.append(page)
            
    return filtered

def _ensure_hyperlink_style(doc: Document):
    styles = doc.styles
    try:
        styles["Hyperlink"] = styles["Hyperlink"]
        return
    except KeyError:
        pass
    style_elem = OxmlElement("w:style")
    style_elem.set(qn("w:type"), "character")
    style_elem.set(qn("w:styleId"), "Hyperlink")
    name_elem = OxmlElement("w:name")
    name_elem.set(qn("w:val"), "Hyperlink")
    style_elem.append(name_elem)
    rPr = OxmlElement("w:rPr")
    color = OxmlElement("w:color")
    color.set(qn("w:val"), "0563C1")
    rPr.append(color)
    u = OxmlElement("w:u")
    u.set(qn("w:val"), "single")
    rPr.append(u)
    style_elem.append(rPr)
    doc.styles.element.append(style_elem)

def _resolve_href(href: str) -> str:
    if not href: return ""
    if href.startswith("http"): return href
    if href.startswith("/confluence"):
        return BASE_URL.split("/confluence")[0] + href
    if href.startswith("/"):
        return BASE_URL.rstrip("/") + href
    return href

def _sanitize_url(url: str) -> str:
    try:
        parsed = urlparse(url)
        clean_path = quote(parsed.path, safe="/-._~:/?#[]@!$&'()*+,;=")
        return urlunparse((parsed.scheme, parsed.netloc, clean_path, parsed.params, parsed.query, parsed.fragment))
    except Exception:
        return ""

def _fetch_image(url: str) -> io.BytesIO | None:
    if not url or url.startswith("data:"): return None
    try:
        r = requests.get(url, headers=HEADERS, verify=CERT_PATH, timeout=10, stream=True)
        r.raise_for_status()
        return io.BytesIO(r.content)
    except Exception:
        return None

def _register_link(display_text: str, url: str):
    if url:
        _LINK_REGISTRY.append((_CURRENT_PAGE_TITLE, display_text.strip(), url))

def _add_hyperlink(paragraph, url: str, display_text: str):
    if not url or not display_text: return
    display_text = _sanitize_text(display_text)
    if not display_text: return
    url = _sanitize_url(url)
    if not url:
        paragraph.add_run(display_text)
        return
    _register_link(display_text, url)
    part = paragraph.part
    try:
        r_id = part.relate_to(url, "http://schemas.openxmlformats.org/officeDocument/2006/relationships/hyperlink", is_external=True)
    except Exception:
        paragraph.add_run(display_text)
        return
    hyperlink = OxmlElement("w:hyperlink")
    hyperlink.set(qn("w:rId"), r_id)
    run_elem = OxmlElement("w:r")
    rPr = OxmlElement("w:rPr")
    rStyle = OxmlElement("w:rStyle")
    rStyle.set(qn("w:val"), "Hyperlink")
    rPr.append(rStyle)
    run_elem.append(rPr)
    t = OxmlElement("w:t")
    t.text = display_text
    if display_text != display_text.strip():
        t.set(qn("xml:space"), "preserve")
    run_elem.append(t)
    hyperlink.append(run_elem)
    paragraph._p.append(hyperlink)

def _add_text_runs(paragraph, node, include_images: bool):
    if isinstance(node, NavigableString):
        text = _sanitize_text(str(node))
        if not text: return
        run = paragraph.add_run(text)
        for parent in node.parents:
            if not isinstance(parent, Tag): break
            n = parent.name
            if n in ("strong", "b"): run.bold = True
            if n in ("em", "i"): run.italic = True
            if n == "u": run.underline = True
            if n in ("code", "tt"):
                run.font.name = "Courier New"
                run.font.size = Pt(10)
            if n in ("p", "li", "td", "th", "div", "body"): break
    elif isinstance(node, Tag):
        if node.name == "img":
            if not include_images:
                # User toggled off images to maximize AI performance
                run = paragraph.add_run("[Image omitted to optimize AI processing speed]")
                run.italic = True
                run.font.color.rgb = RGBColor(0x76, 0x76, 0x76)
                return
            src = node.get("src", "").strip()
            if src:
                img_stream = _fetch_image(_resolve_href(src))
                if img_stream:
                    try:
                        paragraph.add_run().add_picture(img_stream, width=Inches(5.5))
                    except Exception:
                        pass
        elif node.name == "a":
            href = node.get("href", "").strip()
            display = node.get_text(strip=False)
            if href and display.strip():
                _add_hyperlink(paragraph, _resolve_href(href), display)
            else:
                for child in node.children:
                    _add_text_runs(paragraph, child, include_images)
        else:
            for child in node.children:
                _add_text_runs(paragraph, child, include_images)

def _process_table(doc: Document, table_tag: Tag, include_images: bool):
    rows_html = table_tag.find_all("tr")
    if not rows_html: return
    first_row_cells = rows_html[0].find_all(["td", "th"])
    n_cols = sum(int(c.get("colspan", 1)) for c in first_row_cells)
    if n_cols == 0: return
    table = doc.add_table(rows=0, cols=n_cols)
    table.style = "Table Grid"
    for tr in rows_html:
        cells_html = tr.find_all(["td", "th"])
        row = table.add_row()
        col_idx = 0
        for cell_html in cells_html:
            if col_idx >= n_cols: break
            cell = row.cells[col_idx]
            p = cell.paragraphs[0]
            _add_text_runs(p, cell_html, include_images)
            if cell_html.name == "th":
                for run in p.runs: run.bold = True
            col_idx += int(cell_html.get("colspan", 1))

def _process_list(doc: Document, list_tag: Tag, level: int = 0, include_images: bool = True):
    ordered = list_tag.name == "ol"
    for li in list_tag.find_all("li", recursive=False):
        p = doc.add_paragraph(style="List Number" if ordered else "List Bullet")
        p.paragraph_format.left_indent = Inches(0.25 * (level + 1))
        for child in li.children:
            if isinstance(child, NavigableString):
                cleaned = _sanitize_text(str(child))
                if cleaned.strip(): p.add_run(cleaned)
            elif isinstance(child, Tag) and child.name not in ("ul", "ol"):
                _add_text_runs(p, child, include_images)
        for nested in li.find_all(["ul", "ol"], recursive=False):
            _process_list(doc, nested, level + 1, include_images)

def _process_element(doc: Document, element: Tag, include_images: bool) -> bool:
    tag = element.name
    if tag in ("h1", "h2", "h3", "h4", "h5", "h6"):
        p = doc.add_paragraph(style=f"Heading {min(int(tag[1]), 4)}")
        _add_text_runs(p, element, include_images)
        return True
    if tag == "p":
        if element.get_text(strip=True) or element.find("img"):
            p = doc.add_paragraph()
            _add_text_runs(p, element, include_images)
        return True
    if tag == "img":
        if not include_images:
            p = doc.add_paragraph("[Image omitted to optimize AI processing speed]")
            p.runs[0].italic = True
            p.runs[0].font.color.rgb = RGBColor(0x76, 0x76, 0x76)
            return True
        src = element.get("src", "").strip()
        if src:
            img_stream = _fetch_image(_resolve_href(src))
            if img_stream:
                p = doc.add_paragraph()
                p.alignment = WD_ALIGN_PARAGRAPH.CENTER
                try: p.add_run().add_picture(img_stream, width=Inches(6.0))
                except Exception: pass
        return True
    if tag == "pre":
        p = doc.add_paragraph(style="No Spacing")
        run = p.add_run(_sanitize_text(element.get_text()))
        run.font.name = "Courier New"
        run.font.size = Pt(9)
        return True
    if tag in ("ul", "ol"):
        _process_list(doc, element, level=0, include_images=include_images)
        return True
    if tag == "table":
        _process_table(doc, table_tag=element, include_images=include_images)
        doc.add_paragraph()
        return True
    if tag == "hr":
        p = doc.add_paragraph()
        pPr = p._p.get_or_add_pPr()
        pBdr = OxmlElement("w:pBdr")
        bottom = OxmlElement("w:bottom")
        bottom.set(qn("w:val"), "single")
        bottom.set(qn("w:sz"), "6")
        bottom.set(qn("w:space"), "1")
        bottom.set(qn("w:color"), "AAAAAA")
        pBdr.append(bottom)
        pPr.append(pBdr)
        return True
    return False

def html_to_docx(doc: Document, html: str, include_images: bool):
    soup = BeautifulSoup(html, "lxml")
    body = soup.find("body") or soup
    queue = list(body.children)
    while queue:
        node = queue.pop(0)
        if isinstance(node, NavigableString):
            text = _sanitize_text(str(node)).strip()
            if text: doc.add_paragraph(text)
            continue
        if not isinstance(node, Tag): continue
        handled = _process_element(doc, node, include_images)
        if not handled:
            queue = list(node.children) + queue

def build_knowledge_base(pages: list[dict], output_file: str = OUTPUT, verbosity: str = "standard", include_images: bool = True):
    global _LINK_REGISTRY, _CURRENT_PAGE_TITLE
    _LINK_REGISTRY = []
    doc = Document()
    _ensure_hyperlink_style(doc)
    space_title = get_space_title(SPACE_KEY)

    # Cover Page
    title = doc.add_heading(_sanitize_text(space_title), level=0)
    title.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph(f"Space: {_sanitize_text(SPACE_KEY)} - auto-generated via Confluence Portal configuration.")
    doc.add_page_break()

    # Page 2: System Guidance Prompt Engineering
    doc.add_heading("AI Processing Parameters", level=1)
    inject_ai_system_prompt(doc, verbosity)
    doc.add_page_break()

    # Table of Contents
    doc.add_heading("Contents", level=1)
    for i, page in enumerate(pages, 1):
        doc.add_paragraph(f"{i}. {_sanitize_text(page['title'])}", style="List Number")
    doc.add_page_break()

    # Process Pages
    for page in pages:
        clean_title = _sanitize_text(page["title"])
        _CURRENT_PAGE_TITLE = clean_title
        doc.add_heading(clean_title, level=1)
        ancestors = page.get("ancestors", [])
        if ancestors:
            path = " > ".join(_sanitize_text(a["title"]) for a in ancestors)
            p = doc.add_paragraph(f"Path: {path}")
            p.runs[0].font.color.rgb = RGBColor(0x70, 0x70, 0x70)
            p.runs[0].font.size = Pt(9)
        html_content = page.get("body", {}).get("export_view", {}).get("value", "")
        if html_content.strip():
            try: html_to_docx(doc, html_content, include_images)
            except Exception as e: doc.add_paragraph(f"[Error processing page contents: {e}]")
        else:
            doc.add_paragraph("[No content]")
        doc.add_page_break()

    # Appendix Setup
    if _LINK_REGISTRY:
        doc.add_heading("Appendix: Links Referenced", level=1)
        current_page = None
        counter = 0
        for page_title, display_text, url in _LINK_REGISTRY:
            if page_title != current_page:
                current_page = page_title
                counter = 0
                doc.add_heading(page_title, level=2)
            counter += 1
            p = doc.add_paragraph()
            p.add_run(f"{counter}. {display_text}: ").bold = True
            p.add_run(url)
    doc.save(output_file)
