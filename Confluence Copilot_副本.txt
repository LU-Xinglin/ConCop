def fetch_space():
    url = f"{CONFLUENCE_URL}/rest/api/content"
    headers = {"Authorization": f"Bearer {TOKEN}", "Accept": "application/json"}
    all_pages, start, limit = [], 0, 50
    
    print(f"开始抓取 Space [{SPACE_KEY}]...")

            "title": p.get("title"),
            "html": p.get("body", {}).get("storage", {}).get("value", "")

        print(f"已获取 {len(all_pages)} 个页面...")
        if "next" not in data.get("_links", {}):
            break
        start += limit
        
    return all_pages

"""
ConCop_export.py
----------------
Pulls every page from a Confluence space and concatenates them into a single
Word document that mirrors Confluence's native Word-export quality.

Key improvement over the original:
  - Uses `body.export_view` instead of `body.storage`.
    export_view returns clean, render-ready HTML (same source Confluence uses
    for its own PDF/Word export), so tables, lists, bold/italic, headings,
    and inline code are all preserved faithfully.
  - Parses that HTML with BeautifulSoup and maps each element to the
    correct python-docx construct (not just raw stripped text).

Requirements:
    pip install requests urllib3 python-docx beautifulsoup4 lxml
"""

import requests
import urllib3
from bs4 import BeautifulSoup, NavigableString, Tag
from docx import Document
from docx.shared import Pt, RGBColor, Inches
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.oxml.ns import qn
from docx.oxml import OxmlElement
import copy

urllib3.disable_warnings()

# ── CONFIG ────────────────────────────────────────────────────────────────────
BASE_URL   = "https://alm-confluence.systems.uk.hsbc/confluence"
HEADERS    = {"Authorization": "Bearer YOUR_TOKEN_HERE"}
CERT_PATH  = r"C:\Users\45451061\certs\hsbc-orca-g2.pem"   # set to False to disable
SPACE_KEY  = "AGRCKB"
OUTPUT     = "knowledge_base.docx"
# ─────────────────────────────────────────────────────────────────────────────


def get_all_pages(space_key: str) -> list[dict]:
    """Fetch all pages in the space with export_view body (mirrors native export)."""
    pages, start, limit = [], 0, 50
    while True:
        url    = f"{BASE_URL}/rest/api/content"
        params = {
            "spaceKey": space_key,
            "type":     "page",
            "status":   "current",
            "expand":   "body.export_view,ancestors",   # <-- KEY CHANGE
            "start":    start,
            "limit":    limit,
        }
        r    = requests.get(url, headers=HEADERS, params=params, verify=CERT_PATH)
        data = r.json()
        pages.extend(data["results"])
        if start + limit >= data["size"]:
            break
        start += limit
    return pages


# ── HTML → DOCX converter ─────────────────────────────────────────────────────

def _apply_inline_style(run, tag: Tag):
    """Bold, italic, underline, code from the tag's own name / parent chain."""
    names = {t.name for t in tag.parents if isinstance(t, Tag)} | {tag.name}
    if names & {"strong", "b"}:
        run.bold = True
    if names & {"em", "i"}:
        run.italic = True
    if names & {"u"}:
        run.underline = True
    if names & {"code", "tt"}:
        run.font.name = "Courier New"
        run.font.size = Pt(10)


def _add_text_runs(paragraph, node):
    """Recursively walk a node, adding text runs with inline formatting."""
    if isinstance(node, NavigableString):
        text = str(node)
        if text:
            run = paragraph.add_run(text)
            # Apply formatting from parent tags
            for parent in node.parents:
                if not isinstance(parent, Tag):
                    break
                n = parent.name
                if n in ("strong", "b"):
                    run.bold = True
                if n in ("em", "i"):
                    run.italic = True
                if n == "u":
                    run.underline = True
                if n in ("code", "tt"):
                    run.font.name = "Courier New"
                    run.font.size = Pt(10)
                if n in ("p", "li", "td", "th", "div"):
                    break  # stop climbing past block elements
    elif isinstance(node, Tag):
        for child in node.children:
            _add_text_runs(paragraph, child)


def _heading_level(tag_name: str) -> int:
    return int(tag_name[1])   # h1→1, h2→2, …


def _process_table(doc: Document, table_tag: Tag):
    """Convert an HTML <table> to a python-docx Table."""
    rows_html = table_tag.find_all("tr")
    if not rows_html:
        return

    # Determine column count from first row
    first_row_cells = rows_html[0].find_all(["td", "th"])
    n_cols = sum(int(c.get("colspan", 1)) for c in first_row_cells)
    if n_cols == 0:
        return

    table = doc.add_table(rows=0, cols=n_cols)
    table.style = "Table Grid"

    for tr in rows_html:
        cells_html = tr.find_all(["td", "th"])
        row = table.add_row()
        col_idx = 0
        for cell_html in cells_html:
            if col_idx >= n_cols:
                break
            cell = row.cells[col_idx]
            # Bold header cells
            is_header = cell_html.name == "th"
            p = cell.paragraphs[0]
            _add_text_runs(p, cell_html)
            if is_header:
                for run in p.runs:
                    run.bold = True
            col_idx += int(cell_html.get("colspan", 1))


def _is_ordered(list_tag: Tag) -> bool:
    return list_tag.name == "ol"


def _process_list(doc: Document, list_tag: Tag, level: int = 0):
    """Recursively process <ul>/<ol>, supporting nesting."""
    ordered = _is_ordered(list_tag)
    for li in list_tag.find_all("li", recursive=False):
        # Add the list item paragraph
        style = "List Number" if ordered else "List Bullet"
        p = doc.add_paragraph(style=style)
        p.paragraph_format.left_indent = Inches(0.25 * (level + 1))

        # Add direct text content (skip nested lists for now)
        for child in li.children:
            if isinstance(child, NavigableString):
                if str(child).strip():
                    p.add_run(str(child))
            elif isinstance(child, Tag) and child.name not in ("ul", "ol"):
                _add_text_runs(p, child)

        # Recurse into nested lists
        for nested in li.find_all(["ul", "ol"], recursive=False):
            _process_list(doc, nested, level + 1)


def html_to_docx(doc: Document, html: str):
    """
    Parse export_view HTML and append its content to `doc`.
    Handles: headings, paragraphs, bold/italic/underline/code,
             ordered & unordered lists (nested), tables, <hr> dividers,
             and <pre> / code blocks.
    """
    soup = BeautifulSoup(html, "lxml")
    body = soup.find("body") or soup

    for element in body.children:
        if isinstance(element, NavigableString):
            continue
        if not isinstance(element, Tag):
            continue

        tag = element.name

        # ── Headings ──────────────────────────────────────────────────────
        if tag in ("h1", "h2", "h3", "h4", "h5", "h6"):
            level = _heading_level(tag)
            heading_style = f"Heading {min(level, 4)}"
            p = doc.add_paragraph(style=heading_style)
            _add_text_runs(p, element)

        # ── Paragraphs ────────────────────────────────────────────────────
        elif tag == "p":
            text = element.get_text(strip=True)
            if not text:
                continue
            p = doc.add_paragraph()
            _add_text_runs(p, element)

        # ── Pre / code block ──────────────────────────────────────────────
        elif tag == "pre":
            p = doc.add_paragraph(style="No Spacing")
            run = p.add_run(element.get_text())
            run.font.name = "Courier New"
            run.font.size = Pt(9)

        # ── Unordered / ordered lists ─────────────────────────────────────
        elif tag in ("ul", "ol"):
            _process_list(doc, element, level=0)

        # ── Tables ────────────────────────────────────────────────────────
        elif tag == "table":
            _process_table(doc, element)
            doc.add_paragraph()   # breathing room after table

        # ── Horizontal rule ───────────────────────────────────────────────
        elif tag == "hr":
            p = doc.add_paragraph()
            pPr = p._p.get_or_add_pPr()
            pBdr = OxmlElement("w:pBdr")
            bottom = OxmlElement("w:bottom")
            bottom.set(qn("w:val"), "single")
            bottom.set(qn("w:sz"), "6")
            bottom.set(qn("w:space"), "1")
            bottom.set(qn("w:color"), "AAAAAA")
            pBdr.append(bottom)
            pPr.append(pBdr)

        # ── Divs — recurse into them ──────────────────────────────────────
        elif tag == "div":
            html_to_docx(doc, str(element))


# ── Main builder ──────────────────────────────────────────────────────────────

def build_knowledge_base(pages: list[dict], output_file: str = OUTPUT):
    doc = Document()

    # Title page
    title = doc.add_heading("Control Knowledge Base", level=0)
    title.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph(f"Space: {SPACE_KEY} — auto-generated via Confluence export_view API")
    doc.add_page_break()

    # Optional: simple table of contents listing
    doc.add_heading("Contents", level=1)
    for i, page in enumerate(pages, 1):
        doc.add_paragraph(f"{i}. {page['title']}", style="List Number")
    doc.add_page_break()

    # Page content
    for page in pages:
        print(f"  Processing: {page['title']}")

        # Page title as H1
        doc.add_heading(page["title"], level=1)

        # Breadcrumb (ancestor path)
        ancestors = page.get("ancestors", [])
        if ancestors:
            path = " › ".join(a["title"] for a in ancestors)
            p = doc.add_paragraph(f"Path: {path}")
            p.runs[0].font.color.rgb = RGBColor(0x70, 0x70, 0x70)
            p.runs[0].font.size = Pt(9)

        # Body — use export_view HTML
        html_content = (
            page.get("body", {})
                .get("export_view", {})
                .get("value", "")
        )

        if html_content.strip():
            html_to_docx(doc, html_content)
        else:
            doc.add_paragraph("[No content]")

        doc.add_page_break()

    doc.save(output_file)
    print(f"\n✅  Saved {len(pages)} pages → {output_file}")


# ── Entry point ───────────────────────────────────────────────────────────────

if __name__ == "__main__":
    print(f"Fetching all pages from space '{SPACE_KEY}' …")
    pages = get_all_pages(SPACE_KEY)
    print(f"Found {len(pages)} pages. Building document …\n")
    build_knowledge_base(pages)

def _apply_inline_style(run, tag: Tag):
    """Bold, italic, underline, code from the tag's own name / parent chain."""
    names = {t.name for t in tag.parents if isinstance(t, Tag)} | {tag.name}
    if names & {"strong", "b"}:
        run.bold = True
    if names & {"em", "i"}:
        run.italic = True
    if names & {"u"}:
        run.underline = True
    if names & {"code", "tt"}:
        run.font.name = "Courier New"
        run.font.size = Pt(10)


def _add_text_runs(paragraph, node):
    """Recursively walk a node, adding text runs with inline formatting."""
    if isinstance(node, NavigableString):
        text = str(node)
        if text:
            run = paragraph.add_run(text)
            # Apply formatting from parent tags
            for parent in node.parents:
                if not isinstance(parent, Tag):
                    break
                n = parent.name
                if n in ("strong", "b"):
                    run.bold = True
                if n in ("em", "i"):
                    run.italic = True
                if n == "u":
                    run.underline = True
                if n in ("code", "tt"):
                    run.font.name = "Courier New"
                    run.font.size = Pt(10)
                if n in ("p", "li", "td", "th", "div"):
                    break  # stop climbing past block elements
    elif isinstance(node, Tag):
        for child in node.children:
            _add_text_runs(paragraph, child)


def _heading_level(tag_name: str) -> int:
    return int(tag_name[1])   # h1→1, h2→2, …


def _process_table(doc: Document, table_tag: Tag):
    """Convert an HTML <table> to a python-docx Table."""
    rows_html = table_tag.find_all("tr")
    if not rows_html:
        return

    # Determine column count from first row
    first_row_cells = rows_html[0].find_all(["td", "th"])
    n_cols = sum(int(c.get("colspan", 1)) for c in first_row_cells)
    if n_cols == 0:
        return

    table = doc.add_table(rows=0, cols=n_cols)
    table.style = "Table Grid"

    for tr in rows_html:
        cells_html = tr.find_all(["td", "th"])
        row = table.add_row()
        col_idx = 0
        for cell_html in cells_html:
            if col_idx >= n_cols:
                break
            cell = row.cells[col_idx]
            # Bold header cells
            is_header = cell_html.name == "th"
            p = cell.paragraphs[0]
            _add_text_runs(p, cell_html)
            if is_header:
                for run in p.runs:
                    run.bold = True
            col_idx += int(cell_html.get("colspan", 1))


def _is_ordered(list_tag: Tag) -> bool:
    return list_tag.name == "ol"


def _process_list(doc: Document, list_tag: Tag, level: int = 0):
    """Recursively process <ul>/<ol>, supporting nesting."""
    ordered = _is_ordered(list_tag)
    for li in list_tag.find_all("li", recursive=False):
        # Add the list item paragraph
        style = "List Number" if ordered else "List Bullet"
        p = doc.add_paragraph(style=style)
        p.paragraph_format.left_indent = Inches(0.25 * (level + 1))

        # Add direct text content (skip nested lists for now)
        for child in li.children:
            if isinstance(child, NavigableString):
                if str(child).strip():
                    p.add_run(str(child))
            elif isinstance(child, Tag) and child.name not in ("ul", "ol"):
                _add_text_runs(p, child)

        # Recurse into nested lists
        for nested in li.find_all(["ul", "ol"], recursive=False):
            _process_list(doc, nested, level + 1)


def _process_element(doc: Document, element: Tag):
    """
    Handle a single block-level element.
    Returns True if handled, False if the caller should descend into children.
    """
    tag = element.name

    # ── Headings ──────────────────────────────────────────────────────────
    if tag in ("h1", "h2", "h3", "h4", "h5", "h6"):
        level = _heading_level(tag)
        p = doc.add_paragraph(style=f"Heading {min(level, 4)}")
        _add_text_runs(p, element)
        return True

    # ── Paragraphs ────────────────────────────────────────────────────────
    if tag == "p":
        if element.get_text(strip=True):
            p = doc.add_paragraph()
            _add_text_runs(p, element)
        return True

    # ── Pre / code block ──────────────────────────────────────────────────
    if tag == "pre":
        p = doc.add_paragraph(style="No Spacing")
        run = p.add_run(element.get_text())
        run.font.name = "Courier New"
        run.font.size = Pt(9)
        return True

    # ── Lists ─────────────────────────────────────────────────────────────
    if tag in ("ul", "ol"):
        _process_list(doc, element, level=0)
        return True

    # ── Tables ────────────────────────────────────────────────────────────
    if tag == "table":
        _process_table(doc, element)
        doc.add_paragraph()
        return True

    # ── Horizontal rule ───────────────────────────────────────────────────
    if tag == "hr":
        p = doc.add_paragraph()
        pPr = p._p.get_or_add_pPr()
        pBdr = OxmlElement("w:pBdr")
        bottom = OxmlElement("w:bottom")
        bottom.set(qn("w:val"), "single")
        bottom.set(qn("w:sz"), "6")
        bottom.set(qn("w:space"), "1")
        bottom.set(qn("w:color"), "AAAAAA")
        pBdr.append(bottom)
        pPr.append(pBdr)
        return True

    # ── Divs / spans / sections — transparent wrappers, descend into them ─
    return False   # caller will queue children


def html_to_docx(doc: Document, html: str):
    """
    Parse export_view HTML and append its content to `doc`.

    Uses an explicit queue instead of recursion so arbitrarily deep
    <div> nesting never hits Python's recursion limit.
    """
    soup = BeautifulSoup(html, "lxml")
    body = soup.find("body") or soup

    # Iterative BFS/DFS over the element tree
    queue = list(body.children)   # process top-level children first
    while queue:
        node = queue.pop(0)

        if isinstance(node, NavigableString):
            # Bare text outside a <p> — wrap it if non-empty
            text = str(node).strip()
            if text:
                doc.add_paragraph(text)
            continue

        if not isinstance(node, Tag):
            continue

        handled = _process_element(doc, node)
        if not handled:
            # Transparent wrapper (div, section, article, span, etc.)
            # Push children to the front of the queue to preserve order
            queue = list(node.children) + queue

def build_knowledge_base(pages: list[dict], output_file: str = OUTPUT):
    doc = Document()

    # Title page
    title = doc.add_heading("Control Knowledge Base", level=0)
    title.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph(f"Space: {SPACE_KEY} — auto-generated via Confluence export_view API")
    doc.add_page_break()

    # Optional: simple table of contents listing
    doc.add_heading("Contents", level=1)
    for i, page in enumerate(pages, 1):
        doc.add_paragraph(f"{i}. {page['title']}", style="List Number")
    doc.add_page_break()

    # Page content
    for page in pages:
        print(f"  Processing: {page['title']}")

        # Page title as H1
        doc.add_heading(page["title"], level=1)

        # Breadcrumb (ancestor path)
        ancestors = page.get("ancestors", [])
        if ancestors:
            path = " › ".join(a["title"] for a in ancestors)
            p = doc.add_paragraph(f"Path: {path}")
            p.runs[0].font.color.rgb = RGBColor(0x70, 0x70, 0x70)
            p.runs[0].font.size = Pt(9)

        # Body — use export_view HTML
        html_content = (
            page.get("body", {})
                .get("export_view", {})
                .get("value", "")
        )

        if html_content.strip():
            html_to_docx(doc, html_content)
        else:
            doc.add_paragraph("[No content]")

        doc.add_page_break()

    doc.save(output_file)
    print(f"\n✅  Saved {len(pages)} pages → {output_file}")

def _add_hyperlink(paragraph, url: str, display_text: str):
    """
    Insert a clickable hyperlink into a paragraph, styled like a browser link
    (blue, underlined) — identical to what Confluence's Word export produces.

    Word stores hyperlinks as relationships on the paragraph's part, referenced
    by a w:hyperlink element wrapping a styled w:r run.
    """
    part = paragraph.part
    r_id = part.relate_to(
        url,
        "http://schemas.openxmlformats.org/officeDocument/2006/relationships/hyperlink",
        is_external=True,
    )

    hyperlink = OxmlElement("w:hyperlink")
    hyperlink.set(qn("r:id"), r_id)

    run_elem = OxmlElement("w:r")

    rPr = OxmlElement("w:rPr")
    rStyle = OxmlElement("w:rStyle")
    rStyle.set(qn("w:val"), "Hyperlink")
    rPr.append(rStyle)
    run_elem.append(rPr)

    t = OxmlElement("w:t")
    t.text = display_text
    if display_text != display_text.strip():
        t.set("{http://www.w3.org/XML/1998/namespace}space", "preserve")
    run_elem.append(t)

    hyperlink.append(run_elem)
    paragraph._p.append(hyperlink)

def _process_table(doc: Document, table_tag: Tag):

    """Recursively process <ul>/<ol>, supporting nesting."""

    for li in list_tag.find_all("li", recursive=False):
        # Add the list item paragraph
        style = "List Number" if ordered else "List Bullet"
        p = doc.add_paragraph(style=style)
        p.paragraph_format.left_indent = Inches(0.25 * (level + 1))

        # Add direct text content (skip nested lists for now)
        for child in li.children:
            if isinstance(child, NavigableString):
                if str(child).strip():
                    p.add_run(str(child))
            elif isinstance(child, Tag) and child.name not in ("ul", "ol"):
                _add_text_runs(p, child)

        # Recurse into nested lists
        for nested in li.find_all(["ul", "ol"], recursive=False):
            _process_list(doc, nested, level + 1)


def _process_element(doc: Document, element: Tag):
    """
    Handle a single block-level element.
    Returns True if handled, False if the caller should descend into children.
    """
    tag = element.name

    # ── Headings ──────────────────────────────────────────────────────────
    if tag in ("h1", "h2", "h3", "h4", "h5", "h6"):
        level = _heading_level(tag)
        p = doc.add_paragraph(style=f"Heading {min(level, 4)}")
        _add_text_runs(p, element)
        return True

    # ── Paragraphs ────────────────────────────────────────────────────────
    if tag == "p":
        if element.get_text(strip=True):
            p = doc.add_paragraph()
            _add_text_runs(p, element)
        return True

    # ── Pre / code block ──────────────────────────────────────────────────
    if tag == "pre":
        p = doc.add_paragraph(style="No Spacing")
        run = p.add_run(element.get_text())
        run.font.name = "Courier New"
        run.font.size = Pt(9)
        return True

    # ── Lists ─────────────────────────────────────────────────────────────
    if tag in ("ul", "ol"):
        _process_list(doc, element, level=0)
        return True

    # ── Tables ────────────────────────────────────────────────────────────
    if tag == "table":
        _process_table(doc, element)
        doc.add_paragraph()
        return True

    # ── Horizontal rule ───────────────────────────────────────────────────
    if tag == "hr":
        p = doc.add_paragraph()
        pPr = p._p.get_or_add_pPr()
        pBdr = OxmlElement("w:pBdr")
        bottom = OxmlElement("w:bottom")
        bottom.set(qn("w:val"), "single")
        bottom.set(qn("w:sz"), "6")
        bottom.set(qn("w:space"), "1")
        bottom.set(qn("w:color"), "AAAAAA")
        pBdr.append(bottom)
        pPr.append(pBdr)
        return True

    # ── Divs / spans / sections — transparent wrappers, descend into them ─
    return False   # caller will queue children


def html_to_docx(doc: Document, html: str):
    """
    Parse export_view HTML and append its content to `doc`.

    Uses an explicit queue instead of recursion so arbitrarily deep
    <div> nesting never hits Python's recursion limit.
    """
    soup = BeautifulSoup(html, "lxml")
    body = soup.find("body") or soup

    # Iterative BFS/DFS over the element tree
    queue = list(body.children)   # process top-level children first
    while queue:
        node = queue.pop(0)

        if isinstance(node, NavigableString):
            # Bare text outside a <p> — wrap it if non-empty
            text = str(node).strip()
            if text:
                doc.add_paragraph(text)
            continue

        if not isinstance(node, Tag):
            continue

        handled = _process_element(doc, node)
        if not handled:
            # Transparent wrapper (div, section, article, span, etc.)
            # Push children to the front of the queue to preserve order
            queue = list(node.children) + queue

# ── Hyperlink XML builder ──────────────────────────────────────────────────────

def _add_hyperlink(paragraph, url: str, display_text: str):
    """
    Insert a properly styled, clickable hyperlink run into a paragraph.
    Blue (#0563C1) + underline, matching Word's native Hyperlink style.
    """
    if not url or not display_text:
        return

    part = paragraph.part
    try:
        r_id = part.relate_to(
            url,
            "http://schemas.openxmlformats.org/officeDocument/2006/relationships/hyperlink",
            is_external=True,
        )
    except Exception as e:
        print(f"  [hyperlink skipped — bad URL] {url!r}: {e}")
        paragraph.add_run(display_text)
        return

    # w:hyperlink element
    hyperlink = OxmlElement("w:hyperlink")
    hyperlink.set(qn("r:id"), r_id)

    # w:r  (the run inside the hyperlink)
    run_elem = OxmlElement("w:r")

    # w:rPr — apply the Hyperlink character style
    rPr = OxmlElement("w:rPr")
    rStyle = OxmlElement("w:rStyle")
    rStyle.set(qn("w:val"), "Hyperlink")
    rPr.append(rStyle)
    run_elem.append(rPr)

    # w:t — the visible text
    t = OxmlElement("w:t")
    t.text = display_text
    # preserve leading/trailing whitespace if present
    if display_text != display_text.strip():
        t.set("{http://www.w3.org/XML/1998/namespace}space", "preserve")
    run_elem.append(t)

    hyperlink.append(run_elem)
    paragraph._p.append(hyperlink)


# ── Inline text/run builder ────────────────────────────────────────────────────

def _add_text_runs(paragraph, node):
    """
    Recursively walk a BS4 node, emitting Word runs with correct inline
    formatting (bold, italic, underline, code) and clickable hyperlinks.
    """
    if isinstance(node, NavigableString):
        text = str(node)
        if not text:
            return
        run = paragraph.add_run(text)
        # Walk up the parent chain to apply inherited inline styles
        for parent in node.parents:
            if not isinstance(parent, Tag):
                break
            n = parent.name
            if n in ("strong", "b"):
                run.bold = True
            if n in ("em", "i"):
                run.italic = True
            if n == "u":
                run.underline = True
            if n in ("code", "tt"):
                run.font.name = "Courier New"
                run.font.size = Pt(10)
            # Stop climbing once we hit a block-level boundary
            if n in ("p", "li", "td", "th", "div", "body"):
                break

    elif isinstance(node, Tag):
        if node.name == "a":
            href    = node.get("href", "").strip()
            display = node.get_text(strip=False)  # preserve spacing inside link text
            classes = node.get("class", [])

            if href and display.strip():
                resolved = _resolve_href(href, classes)
                if resolved:
                    _add_hyperlink(paragraph, resolved, display)
                else:
                    # Unresolvable href — just emit text
                    paragraph.add_run(display)
            else:
                # Named anchor or empty link — emit children as plain text
                for child in node.children:
                    _add_text_runs(paragraph, child)
        else:
            for child in node.children:
                _add_text_runs(paragraph, child)


# ── Table builder ──────────────────────────────────────────────────────────────

def _process_table(doc: Document, table_tag: Tag):
    """Convert an HTML <table> to a python-docx Table, preserving hyperlinks in cells."""
    rows_html = table_tag.find_all("tr")
    if not rows_html:
        return

    first_row_cells = rows_html[0].find_all(["td", "th"])
    n_cols = sum(int(c.get("colspan", 1)) for c in first_row_cells)
    if n_cols == 0:
        return

    table = doc.add_table(rows=0, cols=n_cols)
    table.style = "Table Grid"

    for tr in rows_html:
        cells_html = tr.find_all(["td", "th"])
        row = table.add_row()
        col_idx = 0
        for cell_html in cells_html:
            if col_idx >= n_cols:
                break
            cell      = row.cells[col_idx]
            is_header = cell_html.name == "th"
            p         = cell.paragraphs[0]
            # Use _add_text_runs so hyperlinks inside cells are rendered correctly
            _add_text_runs(p, cell_html)
            if is_header:
                for run in p.runs:
                    run.bold = True
            col_idx += int(cell_html.get("colspan", 1))


# ── List builder ───────────────────────────────────────────────────────────────

def _process_list(doc: Document, list_tag: Tag, level: int = 0):
    """Recursively process <ul>/<ol>, supporting nesting and hyperlinks in items."""
    ordered = list_tag.name == "ol"
    for li in list_tag.find_all("li", recursive=False):
        style = "List Number" if ordered else "List Bullet"
        p = doc.add_paragraph(style=style)
        p.paragraph_format.left_indent = Inches(0.25 * (level + 1))

        for child in li.children:
            if isinstance(child, NavigableString):
                if str(child).strip():
                    p.add_run(str(child))
            elif isinstance(child, Tag) and child.name not in ("ul", "ol"):
                _add_text_runs(p, child)

        for nested in li.find_all(["ul", "ol"], recursive=False):
            _process_list(doc, nested, level + 1)


# ── Single element dispatcher ──────────────────────────────────────────────────

def _process_element(doc: Document, element: Tag) -> bool:
    """
    Dispatch a single block element to the right builder.
    Returns True if handled (stop), False if caller should descend into children.
    """
    tag = element.name

    if tag in ("h1", "h2", "h3", "h4", "h5", "h6"):
        level = int(tag[1])
        p = doc.add_paragraph(style=f"Heading {min(level, 4)}")
        _add_text_runs(p, element)
        return True

    if tag == "p":
        if element.get_text(strip=True):
            p = doc.add_paragraph()
            _add_text_runs(p, element)
        return True

    if tag == "pre":
        p = doc.add_paragraph(style="No Spacing")
        run = p.add_run(element.get_text())
        run.font.name = "Courier New"
        run.font.size = Pt(9)
        return True

    if tag in ("ul", "ol"):
        _process_list(doc, element, level=0)
        return True

    if tag == "table":
        _process_table(doc, element)
        doc.add_paragraph()  # breathing room after table
        return True

    if tag == "hr":
        p = doc.add_paragraph()
        pPr = p._p.get_or_add_pPr()
        pBdr = OxmlElement("w:pBdr")
        bottom = OxmlElement("w:bottom")
        bottom.set(qn("w:val"), "single")
        bottom.set(qn("w:sz"), "6")
        bottom.set(qn("w:space"), "1")
        bottom.set(qn("w:color"), "AAAAAA")
        pBdr.append(bottom)
        pPr.append(pBdr)
        return True

    return False  # transparent wrapper — caller will descend into children


# ── Main HTML → DOCX converter ─────────────────────────────────────────────────

def html_to_docx(doc: Document, html: str):
    """
    Parse export_view HTML and append content to `doc`.
    Uses an explicit queue (no recursion) to handle arbitrarily deep <div> nesting.
    """
    soup = BeautifulSoup(html, "lxml")
    body = soup.find("body") or soup
    queue = list(body.children)

    while queue:
        node = queue.pop(0)

        if isinstance(node, NavigableString):
            text = str(node).strip()
            if text:
                doc.add_paragraph(text)
            continue

        if not isinstance(node, Tag):
            continue

        handled = _process_element(doc, node)
        if not handled:
            # Transparent wrapper (div, section, span, etc.) — push children
            queue = list(node.children) + queue


# ── Document builder ───────────────────────────────────────────────────────────

def build_knowledge_base(pages: list[dict], output_file: str = OUTPUT):
    doc = Document()

    # Ensure the Hyperlink character style is defined before any content is added
    _ensure_hyperlink_style(doc)

    # Cover page
    title = doc.add_heading("Control Knowledge Base", level=0)
    title.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph(f"Space: {SPACE_KEY} — auto-generated via Confluence export_view API")
    doc.add_page_break()

    # Table of contents
    doc.add_heading("Contents", level=1)
    for i, page in enumerate(pages, 1):
        doc.add_paragraph(f"{i}. {page['title']}", style="List Number")
    doc.add_page_break()

    # Page content
    for page in pages:
        print(f"  Processing: {page['title']}")

        doc.add_heading(page["title"], level=1)

        # Ancestor breadcrumb
        ancestors = page.get("ancestors", [])
        if ancestors:
            path = " › ".join(a["title"] for a in ancestors)
            p = doc.add_paragraph(f"Path: {path}")
            p.runs[0].font.color.rgb = RGBColor(0x70, 0x70, 0x70)
            p.runs[0].font.size = Pt(9)

        html_content = (
            page.get("body", {})
                .get("export_view", {})
                .get("value", "")
        )

        if html_content.strip():
            html_to_docx(doc, html_content)
        else:
            doc.add_paragraph("[No content]")

        doc.add_page_break()

    doc.save(output_file)
    print(f"\n✅  Saved {len(pages)} pages → {output_file}")

    print(f"Fetching all pages from space '{SPACE_KEY}' …")
    pages = get_all_pages(SPACE_KEY)
    print(f"Found {len(pages)} pages. Building document …\n")
    build_knowledge_base(pages)

# ── Confluence API ─────────────────────────────────────────────────────────────

def get_all_pages(space_key: str) -> list[dict]:
    """Fetch all pages in the space with export_view body."""
    pages, start, limit = [], 0, 50
    while True:
        url = f"{BASE_URL}/rest/api/content"
        params = {
            "spaceKey": space_key,
            "type":     "page",
            "status":   "current",
            "expand":   "body.export_view,ancestors",
            "start":    start,
            "limit":    limit,
        }
        r    = requests.get(url, headers=HEADERS, params=params, verify=CERT_PATH)
        data = r.json()
        pages.extend(data["results"])
        if start + limit >= data["size"]:
            break
        start += limit
    return pages


# ── Hyperlink style helper ─────────────────────────────────────────────────────

def _ensure_hyperlink_style(doc: Document):
    """
    Ensure the 'Hyperlink' character style exists in the document.
    Word's default blank document sometimes omits it — if we reference it
    without defining it first, the hyperlink renders as unstyled plain text.
    """
    styles = doc.styles
    try:
        styles["Hyperlink"]
        return  # already exists
    except KeyError:
        pass

    from docx.oxml.ns import qn
    from docx.oxml import OxmlElement

    # Add the style to the document XML directly
    style_elem = OxmlElement("w:style")
    style_elem.set(qn("w:type"), "character")
    style_elem.set(qn("w:styleId"), "Hyperlink")

    name_elem = OxmlElement("w:name")
    name_elem.set(qn("w:val"), "Hyperlink")
    style_elem.append(name_elem)

    rPr = OxmlElement("w:rPr")

    color = OxmlElement("w:color")
    color.set(qn("w:val"), "0563C1")   # standard Word hyperlink blue
    rPr.append(color)

    u = OxmlElement("w:u")
    u.set(qn("w:val"), "single")
    rPr.append(u)

    style_elem.append(rPr)
    doc.styles.element.append(style_elem)


# ── URL resolution ─────────────────────────────────────────────────────────────

def _resolve_href(href: str, classes: list) -> str:
    """
    Normalise a Confluence href to an absolute URL.

    Patterns seen in export_view HTML:
      Type 1 — external link:  class="external-link", href="https://..."
      Type 2 — internal abs.:  href="https://alm-confluence.systems.uk.hsbc/..."
      Type 3 — internal rel.:  href="/confluence/spaces/..." or
                                href="/confluence/display/~43423200"
      Type 4 — other relative: href="/..."
    """
    if not href:
        return ""

    # Types 1 & 2: already absolute
    if href.startswith("http"):
        return href

    # Type 3: starts with /confluence  →  prepend bare domain only
    if href.startswith("/confluence"):
        domain = BASE_URL.split("/confluence")[0]   # "https://alm-confluence.systems.uk.hsbc"
        return domain + href

    # Type 4: any other relative path  →  prepend full BASE_URL
    if href.startswith("/"):
        return BASE_URL.rstrip("/") + href

    # Anything else (mailto:, javascript:, anchors, etc.) — return as-is
    return href


# ── Link collector for appendix ────────────────────────────────────────────────
# Holds (page_title, display_text, url) for every hyperlink encountered.
# Reset at the start of build_knowledge_base, read at the end for the appendix.
_LINK_REGISTRY: list[tuple[str, str, str]] = []
_CURRENT_PAGE_TITLE: str = ""


def _register_link(display_text: str, url: str):
    """Record a link against the page currently being processed."""
    if url:
        _LINK_REGISTRY.append((_CURRENT_PAGE_TITLE, display_text.strip(), url))


# ── Hyperlink XML builder ──────────────────────────────────────────────────────

def _add_hyperlink(paragraph, url: str, display_text: str):
    """
    Insert a properly styled, clickable hyperlink run into a paragraph.
    Blue (#0563C1) + underline, matching Word's native Hyperlink style.
    Also registers the link in _LINK_REGISTRY so it can be listed in an appendix.
    """
    if not url or not display_text:
        return

    _register_link(display_text, url)

    part = paragraph.part
    try:
        r_id = part.relate_to(
            url,
            "http://schemas.openxmlformats.org/officeDocument/2006/relationships/hyperlink",
            is_external=True,
        )
    except Exception as e:
        print(f"  [hyperlink skipped — bad URL] {url!r}: {e}")
        paragraph.add_run(display_text)
        return

    # w:hyperlink element
    hyperlink = OxmlElement("w:hyperlink")
    hyperlink.set(qn("r:id"), r_id)

    # w:r  (the run inside the hyperlink)
    run_elem = OxmlElement("w:r")

    # w:rPr — apply the Hyperlink character style
    rPr = OxmlElement("w:rPr")
    rStyle = OxmlElement("w:rStyle")
    rStyle.set(qn("w:val"), "Hyperlink")
    rPr.append(rStyle)
    run_elem.append(rPr)

    # w:t — the visible text
    t = OxmlElement("w:t")
    t.text = display_text
    # preserve leading/trailing whitespace if present
    if display_text != display_text.strip():
        t.set("{http://www.w3.org/XML/1998/namespace}space", "preserve")
    run_elem.append(t)

    hyperlink.append(run_elem)
    paragraph._p.append(hyperlink)


# ── Inline text/run builder ────────────────────────────────────────────────────

def _add_text_runs(paragraph, node):
    """
    Recursively walk a BS4 node, emitting Word runs with correct inline
    formatting (bold, italic, underline, code) and clickable hyperlinks.
    """
    if isinstance(node, NavigableString):
        text = str(node)
        if not text:
            return
        run = paragraph.add_run(text)
        # Walk up the parent chain to apply inherited inline styles
        for parent in node.parents:
            if not isinstance(parent, Tag):
                break
            n = parent.name
            if n in ("strong", "b"):
                run.bold = True
            if n in ("em", "i"):
                run.italic = True
            if n == "u":
                run.underline = True
            if n in ("code", "tt"):
                run.font.name = "Courier New"
                run.font.size = Pt(10)
            # Stop climbing once we hit a block-level boundary
            if n in ("p", "li", "td", "th", "div", "body"):
                break

    elif isinstance(node, Tag):
        if node.name == "a":
            href    = node.get("href", "").strip()
            display = node.get_text(strip=False)  # preserve spacing inside link text
            classes = node.get("class", [])

            if href and display.strip():
                resolved = _resolve_href(href, classes)
                if resolved:
                    _add_hyperlink(paragraph, resolved, display)
                else:
                    # Unresolvable href — just emit text
                    paragraph.add_run(display)
            else:
                # Named anchor or empty link — emit children as plain text
                for child in node.children:
                    _add_text_runs(paragraph, child)
        else:
            for child in node.children:
                _add_text_runs(paragraph, child)


# ── Table builder ──────────────────────────────────────────────────────────────

def _process_table(doc: Document, table_tag: Tag):
    """Convert an HTML <table> to a python-docx Table, preserving hyperlinks in cells."""
    rows_html = table_tag.find_all("tr")
    if not rows_html:
        return

    first_row_cells = rows_html[0].find_all(["td", "th"])
    n_cols = sum(int(c.get("colspan", 1)) for c in first_row_cells)
    if n_cols == 0:
        return

    table = doc.add_table(rows=0, cols=n_cols)
    table.style = "Table Grid"

    for tr in rows_html:
        cells_html = tr.find_all(["td", "th"])
        row = table.add_row()
        col_idx = 0
        for cell_html in cells_html:
            if col_idx >= n_cols:
                break
            cell      = row.cells[col_idx]
            is_header = cell_html.name == "th"
            p         = cell.paragraphs[0]
            # Use _add_text_runs so hyperlinks inside cells are rendered correctly
            _add_text_runs(p, cell_html)
            if is_header:
                for run in p.runs:
                    run.bold = True
            col_idx += int(cell_html.get("colspan", 1))


# ── List builder ───────────────────────────────────────────────────────────────

def _process_list(doc: Document, list_tag: Tag, level: int = 0):
    """Recursively process <ul>/<ol>, supporting nesting and hyperlinks in items."""
    ordered = list_tag.name == "ol"
    for li in list_tag.find_all("li", recursive=False):
        style = "List Number" if ordered else "List Bullet"
        p = doc.add_paragraph(style=style)
        p.paragraph_format.left_indent = Inches(0.25 * (level + 1))

        for child in li.children:
            if isinstance(child, NavigableString):
                if str(child).strip():
                    p.add_run(str(child))
            elif isinstance(child, Tag) and child.name not in ("ul", "ol"):
                _add_text_runs(p, child)

        for nested in li.find_all(["ul", "ol"], recursive=False):
            _process_list(doc, nested, level + 1)


# ── Single element dispatcher ──────────────────────────────────────────────────

def _process_element(doc: Document, element: Tag) -> bool:
    """
    Dispatch a single block element to the right builder.
    Returns True if handled (stop), False if caller should descend into children.
    """
    tag = element.name

    if tag in ("h1", "h2", "h3", "h4", "h5", "h6"):
        level = int(tag[1])
        p = doc.add_paragraph(style=f"Heading {min(level, 4)}")
        _add_text_runs(p, element)
        return True

    if tag == "p":
        if element.get_text(strip=True):
            p = doc.add_paragraph()
            _add_text_runs(p, element)
        return True

    if tag == "pre":
        p = doc.add_paragraph(style="No Spacing")
        run = p.add_run(element.get_text())
        run.font.name = "Courier New"
        run.font.size = Pt(9)
        return True

    if tag in ("ul", "ol"):
        _process_list(doc, element, level=0)
        return True

    if tag == "table":
        _process_table(doc, element)
        doc.add_paragraph()  # breathing room after table
        return True

    if tag == "hr":
        p = doc.add_paragraph()
        pPr = p._p.get_or_add_pPr()
        pBdr = OxmlElement("w:pBdr")
        bottom = OxmlElement("w:bottom")
        bottom.set(qn("w:val"), "single")
        bottom.set(qn("w:sz"), "6")
        bottom.set(qn("w:space"), "1")
        bottom.set(qn("w:color"), "AAAAAA")
        pBdr.append(bottom)
        pPr.append(pBdr)
        return True

    return False  # transparent wrapper — caller will descend into children


# ── Main HTML → DOCX converter ─────────────────────────────────────────────────

def html_to_docx(doc: Document, html: str):
    """
    Parse export_view HTML and append content to `doc`.
    Uses an explicit queue (no recursion) to handle arbitrarily deep <div> nesting.
    """
    soup = BeautifulSoup(html, "lxml")
    body = soup.find("body") or soup
    queue = list(body.children)

    while queue:
        node = queue.pop(0)

        if isinstance(node, NavigableString):
            text = str(node).strip()
            if text:
                doc.add_paragraph(text)
            continue

        if not isinstance(node, Tag):
            continue

        handled = _process_element(doc, node)
        if not handled:
            # Transparent wrapper (div, section, span, etc.) — push children
            queue = list(node.children) + queue

    print(f"\n✅  Saved {len(pages)} pages → {output_file}")
    print(f"   Appendix contains {len(_LINK_REGISTRY)} links across {len(set(p for p,_,_ in _LINK_REGISTRY))} pages.")

"""
app.py  —  Confluence Knowledge Base Portal
--------------------------------------------
Runs a lightweight Flask web server that lets colleagues trigger a
Confluence space export and download the resulting Word document.

Usage:
    pip install flask requests urllib3 python-docx beautifulsoup4 lxml
    python app.py

Then open http://<your-machine-ip>:5000 in a browser on the internal network.
"""

import os
import uuid
import threading
from datetime import datetime
from flask import Flask, render_template, request, jsonify, send_file, abort

# Import the export logic from ConCop_export.py (same folder)
import ConCop_export as exporter

app = Flask(__name__)

# ── Output folder (all generated docs land here) ───────────────────────────────
OUTPUT_DIR = os.path.join(os.path.dirname(__file__), "generated")
os.makedirs(OUTPUT_DIR, exist_ok=True)

# ── In-memory job tracker ──────────────────────────────────────────────────────
# { job_id: { "status": "running"|"done"|"error", "file": path, "message": str } }
jobs: dict[str, dict] = {}
jobs_lock = threading.Lock()


def run_export(job_id: str, space_key: str, token: str, cert_path: str):
    """
    Background thread: runs the Confluence export and updates the job record.
    """
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    filename  = f"{space_key}_{timestamp}.docx"
    filepath  = os.path.join(OUTPUT_DIR, filename)

    try:
        # Patch the exporter's runtime config for this request
        exporter.BASE_URL   = exporter.BASE_URL          # keep default
        exporter.HEADERS    = {"Authorization": f"Bearer {token}"}
        exporter.CERT_PATH  = cert_path if cert_path else False
        exporter.SPACE_KEY  = space_key
        exporter.OUTPUT     = filepath

        pages = exporter.get_all_pages(space_key)
        if not pages:
            raise ValueError(f"No pages found in space '{space_key}'. "
                             "Check the space key and token permissions.")

        exporter.build_knowledge_base(pages, output_file=filepath)

        with jobs_lock:
            jobs[job_id] = {
                "status":  "done",
                "file":    filepath,
                "filename": filename,
                "pages":   len(pages),
                "links":   len(exporter._LINK_REGISTRY),
                "message": f"Export complete — {len(pages)} pages, "
                           f"{len(exporter._LINK_REGISTRY)} links collected.",
            }

    except Exception as e:
        with jobs_lock:
            jobs[job_id] = {
                "status":  "error",
                "message": str(e),
            }


# ── Routes ─────────────────────────────────────────────────────────────────────

@app.route("/")
def index():
    return render_template("index.html")


@app.route("/export", methods=["POST"])
def start_export():
    """
    Accepts form data, kicks off a background export job, returns the job ID.
    """
    space_key = request.form.get("space_key", "").strip().upper()
    token     = request.form.get("token", "").strip()
    cert_path = request.form.get("cert_path", "").strip()

    if not space_key:
        return jsonify({"error": "Space key is required."}), 400
    if not token:
        return jsonify({"error": "Bearer token is required."}), 400

    job_id = str(uuid.uuid4())
    with jobs_lock:
        jobs[job_id] = {"status": "running", "message": "Export started…"}

    thread = threading.Thread(
        target=run_export,
        args=(job_id, space_key, token, cert_path),
        daemon=True,
    )
    thread.start()

    return jsonify({"job_id": job_id})


@app.route("/status/<job_id>")
def job_status(job_id: str):
    """Polled by the front-end every 2 s to check progress."""
    with jobs_lock:
        job = jobs.get(job_id)
    if not job:
        return jsonify({"error": "Job not found."}), 404
    return jsonify(job)


@app.route("/download/<job_id>")
def download(job_id: str):
    """Serves the generated .docx file as a download."""
    with jobs_lock:
        job = jobs.get(job_id)
    if not job or job.get("status") != "done":
        abort(404)
    filepath = job.get("file")
    if not filepath or not os.path.exists(filepath):
        abort(404)
    return send_file(
        filepath,
        as_attachment=True,
        download_name=job.get("filename", "knowledge_base.docx"),
        mimetype="application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    )


# ── Entry point ────────────────────────────────────────────────────────────────

if __name__ == "__main__":
    # host="0.0.0.0" makes it reachable from other machines on the same network
    app.run(host="0.0.0.0", port=5000, debug=False)

# readme V1
# Confluence Knowledge Base Portal

A lightweight internal web portal that exports any Confluence space into a
structured Word document and serves it as a download.

---

## Folder structure

```
confluence_portal/
├── app.py                  ← Flask server
├── ConCop_export.py        ← Export engine (unchanged)
├── requirements.txt
├── README.md
├── templates/
│   └── index.html          ← Front-end UI
└── generated/              ← Auto-created; exported .docx files land here
```

---

## Setup (one time)

```bash
pip install -r requirements.txt
```

---

## Running

```bash
python app.py
```

The server starts on port 5000 and is reachable from any machine on the
same internal network:

```
http://<your-machine-ip>:5000
```

To find your machine IP on Windows:
```
ipconfig
```
Look for "IPv4 Address" under your active network adapter.

---

## What users see

1. Enter the **Confluence Space Key** (e.g. `AGRCKB`)
2. Paste their **Bearer Token** (generated in Confluence → Profile → Personal Access Tokens)
3. Optionally enter the **certificate path** (the `.pem` file for HSBC's internal SSL)
4. Click **Generate Knowledge Base**
5. Wait for the export to complete — the page shows live progress
6. Click **Download Word Document**

---

## Notes

- Generated files are saved in the `generated/` folder with a timestamp
  in the filename, e.g. `AGRCKB_20240701_143022.docx`.
- The server handles one export per request concurrently using background
  threads — multiple colleagues can trigger exports simultaneously.
- Bearer tokens are never stored; they are only held in memory for the
  duration of the export request.
- To make the server always-on (not tied to your laptop), ask IT for a
  Windows VM or server and run `python app.py` on startup via Task Scheduler.

"""
ConCop_export.py
----------------
Pulls every page from a Confluence space and concatenates them into a single
Word document that mirrors Confluence's native Word-export quality.

Requirements:
    pip install requests urllib3 python-docx beautifulsoup4 lxml
"""

import urllib3
from bs4 import BeautifulSoup, NavigableString, Tag
from docx import Document
from docx.shared import Pt, RGBColor, Inches
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.oxml.ns import qn
from docx.oxml import OxmlElement

urllib3.disable_warnings()

# ── CONFIG ─────────────────────────────────────────────────────────────────────
BASE_URL  = "https://alm-confluence.systems.uk.hsbc/confluence"
HEADERS   = {"Authorization": "Bearer YOUR_TOKEN_HERE"}
CERT_PATH = r"C:\Users\45451061\certs\hsbc-orca-g2.pem"  # or False to disable
SPACE_KEY = "AGRCKB"
OUTPUT    = "knowledge_base.docx"

# ── Confluence API ─────────────────────────────────────────────────────────────

def get_space_title(space_key: str) -> str:
    """Fetch the display name of a Confluence space (e.g. 'Payments Control')."""
    url = f"{BASE_URL}/rest/api/space/{space_key}"
    r   = requests.get(url, headers=HEADERS, verify=CERT_PATH)
    return r.json().get("name", space_key)  # fallback to key if name not found


def get_all_pages(space_key: str) -> list[dict]:
    """Fetch all pages in the space with export_view body."""
    pages, start, limit = [], 0, 50
    while True:
        url = f"{BASE_URL}/rest/api/content"
        params = {
            "spaceKey": space_key,
            "type":     "page",
            "status":   "current",
            "expand":   "body.export_view,ancestors",
            "start":    start,
            "limit":    limit,
        }
        r    = requests.get(url, headers=HEADERS, params=params, verify=CERT_PATH)
        data = r.json()
        pages.extend(data["results"])
        if start + limit >= data["size"]:
            break
        start += limit
    return pages


# ── Hyperlink style helper ─────────────────────────────────────────────────────

def _ensure_hyperlink_style(doc: Document):
    """
    Ensure the 'Hyperlink' character style exists in the document.
    Word's default blank document sometimes omits it — if we reference it
    without defining it first, the hyperlink renders as unstyled plain text.
    """
    styles = doc.styles
    try:
        styles["Hyperlink"]
        return  # already exists
    except KeyError:
        pass

    style_elem = OxmlElement("w:style")
    style_elem.set(qn("w:type"), "character")
    style_elem.set(qn("w:styleId"), "Hyperlink")

    name_elem = OxmlElement("w:name")
    name_elem.set(qn("w:val"), "Hyperlink")
    style_elem.append(name_elem)

    rPr = OxmlElement("w:rPr")

    color = OxmlElement("w:color")
    color.set(qn("w:val"), "0563C1")   # standard Word hyperlink blue
    rPr.append(color)

    u = OxmlElement("w:u")
    u.set(qn("w:val"), "single")
    rPr.append(u)

    style_elem.append(rPr)
    doc.styles.element.append(style_elem)


# ── URL resolution ─────────────────────────────────────────────────────────────

def _resolve_href(href: str, classes: list) -> str:
    """
    Normalise a Confluence href to an absolute URL.

    Patterns seen in export_view HTML:
      Type 1 — external link:  class="external-link", href="https://..."
      Type 2 — internal abs.:  href="https://alm-confluence.systems.uk.hsbc/..."
      Type 3 — internal rel.:  href="/confluence/spaces/..." or
                                href="/confluence/display/~43423200"
      Type 4 — other relative: href="/..."
    """
    if not href:
        return ""

    # Types 1 & 2: already absolute
    if href.startswith("http"):
        return href

    # Type 3: starts with /confluence  →  prepend bare domain only
    if href.startswith("/confluence"):
        domain = BASE_URL.split("/confluence")[0]
        return domain + href

    # Type 4: any other relative path  →  prepend full BASE_URL
    if href.startswith("/"):
        return BASE_URL.rstrip("/") + href

    # Anything else (mailto:, javascript:, anchors, etc.) — return as-is
    return href


# ── Link collector for appendix ────────────────────────────────────────────────
_LINK_REGISTRY: list[tuple[str, str, str]] = []
_CURRENT_PAGE_TITLE: str = ""


def _register_link(display_text: str, url: str):
    """Record a link against the page currently being processed."""
    if url:
        _LINK_REGISTRY.append((_CURRENT_PAGE_TITLE, display_text.strip(), url))


# ── Hyperlink XML builder ──────────────────────────────────────────────────────

def _add_hyperlink(paragraph, url: str, display_text: str):
    """
    Insert a properly styled, clickable hyperlink run into a paragraph.
    Blue (#0563C1) + underline, matching Word's native Hyperlink style.
    Also registers the link in _LINK_REGISTRY so it can be listed in an appendix.
    """
    if not url or not display_text:
        return

    _register_link(display_text, url)

    part = paragraph.part
    try:
        r_id = part.relate_to(
            url,
            "http://schemas.openxmlformats.org/officeDocument/2006/relationships/hyperlink",
            is_external=True,
        )
    except Exception as e:
        print(f"  [hyperlink skipped — bad URL] {url!r}: {e}")
        paragraph.add_run(display_text)
        return

    hyperlink = OxmlElement("w:hyperlink")
    hyperlink.set(qn("r:id"), r_id)

    run_elem = OxmlElement("w:r")

    rPr = OxmlElement("w:rPr")
    rStyle = OxmlElement("w:rStyle")
    rStyle.set(qn("w:val"), "Hyperlink")
    rPr.append(rStyle)
    run_elem.append(rPr)

    t = OxmlElement("w:t")
    t.text = display_text
    if display_text != display_text.strip():
        t.set("{http://www.w3.org/XML/1998/namespace}space", "preserve")
    run_elem.append(t)

    hyperlink.append(run_elem)
    paragraph._p.append(hyperlink)


# ── Inline text/run builder ────────────────────────────────────────────────────

def _add_text_runs(paragraph, node):
    """
    Recursively walk a BS4 node, emitting Word runs with correct inline
    formatting (bold, italic, underline, code) and clickable hyperlinks.
    """
    if isinstance(node, NavigableString):
        text = str(node)
        if not text:
            return
        run = paragraph.add_run(text)
        for parent in node.parents:
            if not isinstance(parent, Tag):
                break
            n = parent.name
            if n in ("strong", "b"):
                run.bold = True
            if n in ("em", "i"):
                run.italic = True
            if n == "u":
                run.underline = True
            if n in ("code", "tt"):
                run.font.name = "Courier New"
                run.font.size = Pt(10)
            if n in ("p", "li", "td", "th", "div", "body"):
                break

    elif isinstance(node, Tag):
        if node.name == "a":
            href    = node.get("href", "").strip()
            display = node.get_text(strip=False)
            classes = node.get("class", [])

            if href and display.strip():
                resolved = _resolve_href(href, classes)
                if resolved:
                    _add_hyperlink(paragraph, resolved, display)
                else:
                    paragraph.add_run(display)
            else:
                for child in node.children:
                    _add_text_runs(paragraph, child)
        else:
            for child in node.children:
                _add_text_runs(paragraph, child)


# ── Table builder ──────────────────────────────────────────────────────────────

def _process_table(doc: Document, table_tag: Tag):
    """Convert an HTML <table> to a python-docx Table, preserving hyperlinks in cells."""
    rows_html = table_tag.find_all("tr")
    if not rows_html:
        return

    first_row_cells = rows_html[0].find_all(["td", "th"])
    n_cols = sum(int(c.get("colspan", 1)) for c in first_row_cells)
    if n_cols == 0:
        return

    table = doc.add_table(rows=0, cols=n_cols)
    table.style = "Table Grid"

    for tr in rows_html:
        cells_html = tr.find_all(["td", "th"])
        row = table.add_row()
        col_idx = 0
        for cell_html in cells_html:
            if col_idx >= n_cols:
                break
            cell      = row.cells[col_idx]
            is_header = cell_html.name == "th"
            p         = cell.paragraphs[0]
            _add_text_runs(p, cell_html)
            if is_header:
                for run in p.runs:
                    run.bold = True
            col_idx += int(cell_html.get("colspan", 1))


# ── List builder ───────────────────────────────────────────────────────────────

def _process_list(doc: Document, list_tag: Tag, level: int = 0):
    """Recursively process <ul>/<ol>, supporting nesting and hyperlinks in items."""
    ordered = list_tag.name == "ol"
    for li in list_tag.find_all("li", recursive=False):
        style = "List Number" if ordered else "List Bullet"
        p = doc.add_paragraph(style=style)
        p.paragraph_format.left_indent = Inches(0.25 * (level + 1))

        for child in li.children:
            if isinstance(child, NavigableString):
                if str(child).strip():
                    p.add_run(str(child))
            elif isinstance(child, Tag) and child.name not in ("ul", "ol"):
                _add_text_runs(p, child)

        for nested in li.find_all(["ul", "ol"], recursive=False):
            _process_list(doc, nested, level + 1)


# ── Single element dispatcher ──────────────────────────────────────────────────

def _process_element(doc: Document, element: Tag) -> bool:
    """
    Dispatch a single block element to the right builder.
    Returns True if handled (stop), False if caller should descend into children.
    """
    tag = element.name

    if tag in ("h1", "h2", "h3", "h4", "h5", "h6"):
        level = int(tag[1])
        p = doc.add_paragraph(style=f"Heading {min(level, 4)}")
        _add_text_runs(p, element)
        return True

    if tag == "p":
        if element.get_text(strip=True):
            p = doc.add_paragraph()
            _add_text_runs(p, element)
        return True

    if tag == "pre":
        p = doc.add_paragraph(style="No Spacing")
        run = p.add_run(element.get_text())
        run.font.name = "Courier New"
        run.font.size = Pt(9)
        return True

    if tag in ("ul", "ol"):
        _process_list(doc, element, level=0)
        return True

    if tag == "table":
        _process_table(doc, element)
        doc.add_paragraph()
        return True

    if tag == "hr":
        p = doc.add_paragraph()
        pPr = p._p.get_or_add_pPr()
        pBdr = OxmlElement("w:pBdr")
        bottom = OxmlElement("w:bottom")
        bottom.set(qn("w:val"), "single")
        bottom.set(qn("w:sz"), "6")
        bottom.set(qn("w:space"), "1")
        bottom.set(qn("w:color"), "AAAAAA")
        pBdr.append(bottom)
        pPr.append(pBdr)
        return True

    return False


# ── Main HTML → DOCX converter ─────────────────────────────────────────────────

def html_to_docx(doc: Document, html: str):
    """
    Parse export_view HTML and append content to `doc`.
    Uses an explicit queue (no recursion) to handle arbitrarily deep <div> nesting.
    """
    soup = BeautifulSoup(html, "lxml")
    body = soup.find("body") or soup
    queue = list(body.children)

    while queue:
        node = queue.pop(0)

        if isinstance(node, NavigableString):
            text = str(node).strip()
            if text:
                doc.add_paragraph(text)
            continue

        if not isinstance(node, Tag):
            continue

        handled = _process_element(doc, node)
        if not handled:
            queue = list(node.children) + queue


# ── Document builder ───────────────────────────────────────────────────────────

def build_knowledge_base(pages: list[dict], output_file: str = OUTPUT):
    global _LINK_REGISTRY
    _LINK_REGISTRY = []  # reset for a fresh run

    doc = Document()

    # Ensure the Hyperlink character style is defined before any content is added
    _ensure_hyperlink_style(doc)

    # Fetch the space's display name to use as the document title
    print(f"  Fetching space title for '{SPACE_KEY}' …")
    space_title = get_space_title(SPACE_KEY)

    # Cover page
    title = doc.add_heading(space_title, level=0)
    title.alignment = WD_ALIGN_PARAGRAPH.CENTER
    doc.add_paragraph(f"Space: {SPACE_KEY} — auto-generated via Confluence export_view API")
    doc.add_page_break()

    # Table of contents
    doc.add_heading("Contents", level=1)
    for i, page in enumerate(pages, 1):
        doc.add_paragraph(f"{i}. {page['title']}", style="List Number")
    doc.add_page_break()

    # Page content
    for page in pages:
        print(f"  Processing: {page['title']}")

        global _CURRENT_PAGE_TITLE
        _CURRENT_PAGE_TITLE = page["title"]

        doc.add_heading(page["title"], level=1)

        # Ancestor breadcrumb
        ancestors = page.get("ancestors", [])
        if ancestors:
            path = " › ".join(a["title"] for a in ancestors)
            p = doc.add_paragraph(f"Path: {path}")
            p.runs[0].font.color.rgb = RGBColor(0x70, 0x70, 0x70)
            p.runs[0].font.size = Pt(9)

        html_content = (
            page.get("body", {})
                .get("export_view", {})
                .get("value", "")
        )

        if html_content.strip():
            html_to_docx(doc, html_content)
        else:
            doc.add_paragraph("[No content]")

        doc.add_page_break()

    # ── Appendix: flat list of all links grouped by source page ───────────
    if _LINK_REGISTRY:
        doc.add_heading("Appendix: Links Referenced", level=1)
        doc.add_paragraph(
            "The list below contains every hyperlink found in the pages above, "
            "grouped by the page it appeared on, in plain-text form."
        )

        current_page = None
        counter = 0
        for page_title, display_text, url in _LINK_REGISTRY:

def get_all_pages(space_key: str, limit: int = 0) -> list[dict]:
    """
    Fetch pages from the space with export_view body.
    limit: max number of pages to fetch. 0 means fetch all.
    """
    pages, start, batch = [], 0, 50
    while True:
        url = f"{BASE_URL}/rest/api/content"
        params = {
            "spaceKey": space_key,
            "type":     "page",
            "status":   "current",
            "expand":   "body.export_view,ancestors",
            "start":    start,
            "limit":    batch,
        }
        r    = requests.get(url, headers=HEADERS, params=params, verify=CERT_PATH)
        data = r.json()
        pages.extend(data["results"])

        # Stop if we've hit the user's page limit
        if limit > 0 and len(pages) >= limit:
            pages = pages[:limit]
            break

        # Stop if Confluence has no more pages
        if start + batch >= data["size"]:
            break

        start += batch
    return pages
