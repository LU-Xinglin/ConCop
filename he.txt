def load_and_deduplicate_data(filepath: str, col_name: str) -> pd.Series:
    """Loads Excel data, removes empty rows, and deduplicates identical issue descriptions."""
    print(f"Loading data from '{filepath}'...")
    df = pd.read_excel(filepath)

    # 1. Drop rows where the description is missing
    df = df.dropna(subset=[col_name])

    # 2. Convert to string and strip leading/trailing whitespace
    df[col_name] = df[col_name].astype(str).str.strip()

    # 3. Filter out rows that became empty strings after stripping
    df = df[df[col_name] != ""]

    # 4. Deduplicate: Isolates unique issues by dropping repeated impact rows
    initial_rows = len(df)
    df = df.drop_duplicates(subset=[col_name], keep="first")
    unique_issues = len(df)

    print(f"--- Data Ingestion Summary ---")
    print(f"Total impact rows read : {initial_rows}")
    print(f"Unique issues isolated : {unique_issues}")
    print(f"Rows removed (duplicates): {initial_rows - unique_issues}\n")

    return df[col_name]


def preprocess_text(text_series: pd.Series) -> list:
    """Tokenizes, removes stopwords/noise, and lemmatizes text."""
    print("Preprocessing and tokenizing descriptions...")
    stop_words = set(stopwords.words("english"))

    # Add high-frequency operational boilerplate words here as you discover them
    custom_stops = {
        "issue",
        "error",
        "event",
        "impact",
        "incident",
        "occurred",
        "failed",
        "failure",
    }
    stop_words.update(custom_stops)

    lemmatizer = WordNetLemmatizer()
    processed_docs = []

    for text in text_series:
        # Remove email addresses, file paths, URLs, and special characters
        text = re.sub(r"\S*@\S*\s?", "", text)
        text = re.sub(r"http\S+", "", text)
        text = re.sub(r"[^a-zA-Z\s]", " ", text)

        # Tokenize into words (converts to lowercase, ignores words < 3 chars)
        tokens = simple_preprocess(text, deacc=True, min_len=3)

        # Remove stopwords and reduce words to their root form (lemmatization)
        clean_tokens = [
            lemmatizer.lemmatize(token)
            for token in tokens
            if token not in stop_words
        ]

        if clean_tokens:  # Only append non-empty document arrays
            processed_docs.append(clean_tokens)

    return processed_docs


def train_and_evaluate_lda(
    dictionary, corpus, texts, start_k=2, max_k=16, step=2
) -> tuple:
    """Trains LDA models across a range of topic counts and finds the best Cv Coherence score."""
    coherence_scores = []
    models = []

    print(f"Searching for optimal topic count between k={start_k} and k={max_k}...")

    for num_topics in range(start_k, max_k + 1, step):
        # Initialize and train LDA model
        model = LdaModel(
            corpus=corpus,
            id2word=dictionary,
            num_topics=num_topics,
            random_state=42,  # Fixed seed for reproducibility
            update_every=1,
            chunksize=100,
            passes=15,  # Increased passes slightly for deduplicated data
            alpha="auto",
            per_word_topics=True,
        )
        models.append(model)

        # Calculate Cv Coherence Score
        coherence_model = CoherenceModel(
            model=model, texts=texts, dictionary=dictionary, coherence="c_v"
        )
        score = coherence_model.get_coherence()
        coherence_scores.append(score)

        print(f"  -> k={num_topics:02d} topics | Coherence (C_v): {score:.4f}")

    # Locate the best model based on the highest Cv score
    best_idx = coherence_scores.index(max(coherence_scores))
    best_model = models[best_idx]
    best_k = start_k + (best_idx * step)

    print(
        f"\nOptimal Model Selected: k={best_k} topics (Coherence = {max(coherence_scores):.4f})"
    )
    return best_model, best_k

"""Create a review workbook of issue descriptions that may be Spanish.

This is intentionally a conservative, offline heuristic.  It flags rows for
human review; it does not delete or translate anything.
"""

from __future__ import annotations

import argparse
import re
from pathlib import Path

import pandas as pd


# Common Spanish function words are much more reliable language signals than
# business terms such as "banco" or "pago", which could appear in a name.
SPANISH_MARKERS = {
    "al",
    "como",
    "con",
    "del",
    "desde",
    "el",
    "en",
    "entre",
    "es",
    "esta",
    "este",
    "la",
    "las",
    "lo",
    "los",
    "más",
    "para",
    "pero",
    "por",
    "que",
    "se",
    "sin",
    "sobre",
    "su",
    "sus",
    "un",
    "una",
    "y",
}

WORD_PATTERN = re.compile(r"[a-záéíóúüñ]+", flags=re.IGNORECASE)


def spanish_markers(text: object) -> list[str]:
    """Return the distinct Spanish marker words found in a description."""
    if pd.isna(text):
        return []

    words = set(WORD_PATTERN.findall(str(text).casefold()))
    return sorted(words & SPANISH_MARKERS)


def build_report(input_path: Path, column: str, output_path: Path) -> None:
    dataframe = pd.read_excel(input_path)

    if column not in dataframe.columns:
        available = ", ".join(map(str, dataframe.columns))
        raise ValueError(
            f"Column {column!r} was not found. Available columns: {available}"
        )

    report = dataframe.copy()
    marker_lists = report[column].apply(spanish_markers)
    report["spanish_marker_count"] = marker_lists.str.len()
    report["spanish_markers_found"] = marker_lists.apply(", ".join)
    # Two or more function words gives a much lower false-positive rate for
    # short incident descriptions than flagging every isolated word such as
    # "para" or "del".
    report["likely_spanish"] = report["spanish_marker_count"] >= 2

    flagged = report.loc[report["likely_spanish"]].copy()
    summary = pd.DataFrame(
        {
            "metric": [
                "Input rows",
                "Likely Spanish rows",
                "Rows not flagged",
                "Detection rule",
            ],
            "value": [
                len(report),
                len(flagged),
                len(report) - len(flagged),
                "At least two Spanish function-word markers",
            ],
        }
    )

    with pd.ExcelWriter(output_path, engine="openpyxl") as writer:
        summary.to_excel(writer, sheet_name="Summary", index=False)
        flagged.to_excel(writer, sheet_name="Likely Spanish", index=False)
        report.to_excel(writer, sheet_name="All rows", index=False)

        for worksheet in writer.sheets.values():
            worksheet.freeze_panes = "A2"
            worksheet.auto_filter.ref = worksheet.dimensions
            for column_cells in worksheet.columns:
                width = min(
                    max(len(str(cell.value or "")) for cell in column_cells) + 2,
                    60,
                )
                worksheet.column_dimensions[column_cells[0].column_letter].width = width

    print(f"Rows read: {len(report)}")
    print(f"Likely Spanish rows: {len(flagged)}")
    print(f"Review workbook written to: {output_path.resolve()}")


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(
        description="Flag likely Spanish issue descriptions for manual review."
    )
    parser.add_argument("input_file", type=Path, help="Path to the source Excel workbook")
    parser.add_argument(
        "--column",
        default="Issue Description",
        help="Description column to inspect (default: Issue Description)",
    )
    parser.add_argument(
        "--output",
        type=Path,
        default=Path("spanish_description_review.xlsx"),
        help="Output review workbook (default: spanish_description_review.xlsx)",
    )
    return parser.parse_args()


if __name__ == "__main__":
    arguments = parse_args()
    build_report(arguments.input_file, arguments.column, arguments.output)
