    <script>
        const scanBtn = document.getElementById("scan-btn");
        const configPanel = document.getElementById("config-panel");
        const parentContainer = document.getElementById("parent-container");
        const vaultForm = document.getElementById("vault-form");
        const submitBtn = document.getElementById("submit-btn");
        
        const statusPanel = document.getElementById("status-panel");
        const statusText = document.getElementById("status-text");
        const spinner = document.getElementById("spinner");
        const statsDiv = document.getElementById("stats");
        const statPages = document.getElementById("stat-pages");
        const statLinks = document.getElementById("stat-links");
        const downloadBtn = document.getElementById("download-btn");

        let pollInterval = null;

        function updateRadioUI(input) {
            document.querySelectorAll(".radio-card").forEach(c => c.classList.remove("active"));
            input.parentElement.classList.add("active");
        }

        scanBtn.addEventListener("click", async () => {
            const spaceUrl = document.getElementById("space_url").value.trim();
            const token = document.getElementById("token").value.trim();
            
            if(!spaceUrl || !token) {
                alert("Please complete address parameters and security token inputs before scanning.");
                return;
            }

            scanBtn.disabled = true;
            scanBtn.textContent = "Scanning Space Map...";

            const fData = new FormData();
            fData.append("space_url", spaceUrl);
            fData.append("token", token);

            try {
                const res = await fetch("/scan", { method: "POST", body: fData });
                const data = await res.json();
                if(data.error) throw new Error(data.error);

                // Build child nodes inside UI
                parentContainer.innerHTML = "";
                data.roots.forEach(r => {
                    const lbl = document.createElement("label");
                    lbl.className = "checkbox-item";
                    lbl.innerHTML = `<input type="checkbox" name="selected_parents" value="${r.id}"> <span>${r.title}</span>`;
                    parentContainer.appendChild(lbl);
                });

                configPanel.style.display = "block";
                scanBtn.style.display = "none";
            } catch(e) {
                alert("Scan failed: " + e.message);
                scanBtn.disabled = false;
                scanBtn.textContent = "Scan Target Space";
            }
        });

        vaultForm.addEventListener("submit", async (e) => {
            e.preventDefault();
            submitBtn.disabled = true;
            statusPanel.style.display = "block";
            statusText.className = "status-text";
            statusText.textContent = "Kicking off pipeline engine process execution...";
            spinner.style.display = "block";
            statsDiv.style.display = "none";
            downloadBtn.style.display = "none";

            const payload = new FormData(vaultForm);
            // Append explicit configuration items manually
            payload.append("include_images", document.getElementById("include_images").checked ? "true" : "false");

            let jId;
            try {
                const res = await fetch("/export", { method: "POST", body: payload });
                const data = await res.json();
                if(data.error) throw new Error(data.error);
                jId = data.job_id;
            } catch(err) {
                showError(err.message);
                return;
            }
            pollInterval = setInterval(() => checkStatus(jId), 2000);
        });

        async function checkStatus(jobId) {
            try {
                const res = await fetch(`/status/${jobId}`);
                const data = await res.json();
                if(data.status === "running") {
                    statusText.textContent = data.message || "Running extraction matrix processes...";
                    return;
                }
                clearInterval(pollInterval);
                if(data.status === "done") {
                    spinner.style.display = "none";
                    statusText.className = "status-text success";
                    statusText.textContent = "✓ Content vector document compiled successfully.";
                    statPages.textContent = data.pages ?? "-";
                    statLinks.textContent = data.links ?? "-";
                    statsDiv.style.display = "flex";
                    downloadBtn.href = `/download/${jobId}`;
                    downloadBtn.style.display = "block";
                    submitBtn.disabled = false;
                } else {
                    showError(data.message || "Fatal pipeline error occurred.");
                }
            } catch(e) {
                showError("Connection interrupted during polling sequence.");
            }
        }

        function showError(msg) {
            clearInterval(pollInterval);
            spinner.style.display = "none";
            statusText.className = "status-text error";
            statusText.textContent = "✕ " + msg;
            submitBtn.disabled = false;
        }
    </script>
